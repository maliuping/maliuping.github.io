.globl _start
.type _start,@function

_start: .cfi_startproc .cfi_undefined ra .option push .option norelax la gp, __global_pointer$ .option pop

// Continue primary hart
csrr a0, mhartid
li   a1, PRIM_HART
bne  a0, a1, secondary

// Primary hart
la sp, _stack_top

// Load data section
la a0, _data_lma
la a1, _data
la a2, _edata
bgeu a1, a2, 2f

1: LOAD t0, (a0) STOR t0, (a1) addi a0, a0, REGSIZE addi a1, a1, REGSIZE bltu a1, a2, 1b 2:

// Clear bss section
la a0, _bss
la a1, _ebss
bgeu a0, a1, 2f

1: STOR zero, (a0) addi a0, a0, REGSIZE bltu a0, a1, 1b 2: mde: // argc, argv, envp is 0 li a0, 0 li a1, 0 li a2, 0 jal main 1: wfi j 1b
